House Price Index

Data is sourced from FHFA Monthly Purchase-Only Index and begins in January 1991.

This is a side note. 
url <- "https://www.fhfa.gov/DataTools/Downloads/Documents/HPI/HPI_PO_monthly_hist.xls"
destfile <- "hpi.xls"
curl::curl_download(url, destfile)
hpi <- read_excel(destfile, skip = 3)

hpi_wrangled <- hpi %>% 
  clean_names() %>% 
  slice(-1) %>%   # remove empty row
  rename(date = month) %>%
  mutate_if(is.numeric, round, digits = 2) %>%  # round all numeric columns to 2 digits
  arrange(date) %>% 
  select(date, usa_sa, usa_nsa)

Seasonally Adjusted

plot_time_series(hpi_wrangled, date, usa_sa, 
                 .interactive = FALSE, 
                 .smooth = FALSE, 
                 .title = "US HPI - Seasonally Adjusted")

Not Seasonally Adjusted

plot_time_series(hpi_wrangled, date, usa_nsa, 
                 .interactive = FALSE, 
                 .smooth = FALSE, 
                 .title = "US HPI - Not Seasonally Adjusted")
